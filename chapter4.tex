\section{Exponential Functions}
\begin{equation}
\label{amn}
a^m . a^n  = a^{m+n}
\end{equation}
\begin{equation}
\label{am/n}
\frac{a^m}{a^n}  = a^{m-n}
\end{equation}
\begin{equation}
\label{a^mn}
({a^m})^{n} = a^{mn}
\end{equation}
\begin{equation}
\label{am_equal_an}
a^m = a^n \implies m = n
\end{equation}
\begin{equation}
\label{atominusn}
a^{-n} = \frac{1}{a^n}
\end{equation}
